# Untuk menjalankan program ini ketik:
- npm run dev

# Inisialisasi Prisma ke direktori saat ini ketik:
- npx prisma init

# Untuk implementasi skema tabel dan relasinya ke database ketik:
- npx prisma generate
- npx prisma db push

# Untuk melihat visualisasi database dari aplikasi Prisma ketik:
- npx prisma studio
